1. Объясните почему код даёт именно такие результаты?
Подсказка. Чтобы лучше разобраться возьмите этот код и запустите в отладчике в 
браузере со включенными точками остановки.
//пример 1
let a = 1, b = 1, c, d; 
c = ++a; //используем инкремент в префиксе, сначала делаем инкримент а (++а), после полученнное значение 2 кладем в переменную с 
alert(c); // ответ: 2

//пример 2
d = b++; //ситуация обратная предыдущей. Используем инкримент в постфиксе, сначала d = b , затем b++
alert(d); //ответ: 1

//пример 3
c = 2 + ++a; //++a = 2 + 1 (в первом примере инкремент а уже был взят, поэтому а = 2), след с = 2 + 3 = 5
alert(c); //ответ: 5

//пример 4
d = 2 + b++; // инкремент - постфикс, след d = 2 + 2 = 4 и только после b++ = 3
alert(d); //ответ: 4

alert(a); //3
alert(b); //3
